### Query with path at entity level and normal property
query data 
{ 
    @traverse(path:"person:Stark/Tony")
    person 
    { 
        nickname
    } 
}

### Query with path at query level and normal property
@traverse(path:"person:Stark/Tony")
query data 
{ 
    person 
    { 
        nickname
    } 
}

### Query with entity path and relative property
query data 
{ 
    @traverse(path:"person:Stark/Tony")
    person 
    { 
        lastname @traverse(path:""\\"") 
    } 
}

### Query with multiple entity paths unioned together returning multiple entries and a normal property
query data 
{ 
    @traverse(path:"person:Vrenken/Peter", mode: "Union")
    @traverse(path:"person:Vrenken/Tanja")
    person 
    { 
        nickname 
    } 
}

### Query with multiple entity paths unioned together returning multiple entries and a normal property
query data 
{ 
    @traverse(path:"person:Vrenken/Peter", mode: "Union")
    @traverse(path:"person:Vrenken/Tanja")
    person 
    { 
        name @id() 
    } 
}


### Query with multiple entity paths intersected together returning two sets of entries. 
# The first for all persons and the second for all locations that match the intersection.
@traverse(path:"location:DE/Berlin//", mode: "Intersect")
@traverse(path:"time:2012//")
@traverse(path:"person:Vrenken/Peter")
{ 
    person @traverse(path:"\persons\")
    { 
        firstname 
        lastname @traverse(path:"\") 
    },
    location @traverse(path:"\locations\") 
    { 
        name @traverse(path:"\\"),
    } 
}
