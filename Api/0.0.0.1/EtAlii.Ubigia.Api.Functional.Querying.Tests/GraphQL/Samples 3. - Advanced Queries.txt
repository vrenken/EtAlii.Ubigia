# This is nested traversal. 
# Developer notice: Currently the traversal at the query level can never be typed.

{ 
    person @nodes(path:"person:Stark/Tony")
    {
        friends @nodes(path:"/Friends")

            name @id()
            nickname
        } 
    }
}
