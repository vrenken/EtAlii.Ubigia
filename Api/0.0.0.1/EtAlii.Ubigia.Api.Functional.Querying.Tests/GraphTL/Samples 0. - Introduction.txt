// Query annotation.
@traverse("Person:Doe/John")
{
    "FirstName",
    "LastName",
    "NickName",
}

@traverse("Person:Doe/John")
{
    FirstName,
    LastName,
    NickName,
}

@traverse("Person:Doe/John")
{
    FirstName @id(),
    LastName @id("\LastName"),
    NickName,
    Friends @traverse("/Friends")  
}
